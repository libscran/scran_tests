27inline bool compare_almost_equal(double left, double right, double tol = 1e-8, bool report = true) {
28 if (left == right) {
29 return true;
30 }
31
32 // Use relative tolerance to check that we get a similar float.
33 // EXPECT_EQ and even EXPECT_DOUBLE_EQ tend to be a bit too stringent when numerical imprecision is expected.
34 // We define the relative difference as '|difference| / |mean|' of the two values, and check that this is below some acceptable error, in this case 1e-8.
35 //
36 // It's worth nothing that I spent a long time using EXPECT_FLOAT_EQ when I really should have been using EXPECT_DOUBLE_EQ.
37 // This would have implicitly used a relative threshold at the 8th significant figure, which is why it allowed tests to pass despite imprecision.
38 const double denom = std::abs(left + right) / 2;
39 double threshold = denom * tol;
40
41 // 'relative difference < tol' is equivalent to '|difference| < threshold'.
42 // However, if '|mean|' is very small, we start to revert back to an exact equality test.
43 // This is not ideal - we DON'T want to fail the test if one value is zero and another is very slightly non-zero.
44 // So we need to ensure that the threshold doesn't get too small, by setting a lower bound similar to the machine epsilon.
45 // (That said, if we're dealing with lots of very-near-zero values, this function won't be appropriate as it'll consider them to be the same.)
46 if (threshold < 1e-15) {
47 threshold = 1e-15;
48 }
49 if (std::abs(left - right) > threshold) {
50 if (report) {
51 EXPECT_TRUE(false) << "mismatch in almost-equal floats (expected " << left << ", got " << right << ")";
52 }
53 return false;
54 }
55
56 return true;
57}
