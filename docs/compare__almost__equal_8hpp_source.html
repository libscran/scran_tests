46inline bool compare_almost_equal(double left, double right, const CompareAlmostEqualParameters& params) {
47 if (left == right) {
48 return true;
49 }
50
51 // Use relative tolerance to check that we get a similar float.
52 // EXPECT_EQ and even EXPECT_DOUBLE_EQ tend to be a bit too stringent when numerical imprecision is expected.
53 // We define the relative difference as '|difference| / |mean|' of the two values, and check that this is below some acceptable error, in this case 1e-8.
54 //
55 // It's worth nothing that I spent a long time using EXPECT_FLOAT_EQ when I really should have been using EXPECT_DOUBLE_EQ.
56 // This would have implicitly used a relative threshold at the 8th significant figure, which is why it allowed tests to pass despite imprecision.
57 const double denom = std::abs(left + right) / 2;
58 double threshold = denom * params.relative_tolerance;
59
60 // 'relative difference < tol' is equivalent to '|difference| < threshold'.
61 // However, if '|mean|' is very small, we start to revert back to an exact equality test.
62 // This is not ideal - we DON'T want to fail the test if one value is zero and another is very slightly non-zero.
63 // So we need to ensure that the threshold doesn't get too small, by setting a lower bound similar to the machine epsilon.
64 // (That said, if we're dealing with lots of very-near-zero values, this function won't be appropriate as it'll consider them to be the same.)
65 if (threshold < params.absolute_tolerance) {
66 threshold = params.absolute_tolerance;
67 }
68 if (std::abs(left - right) > threshold) {
69 if (report) {
70 EXPECT_TRUE(false) << "mismatch in almost-equal floats (expected " << left << ", got " << right << ")";
71 }
72 return false;
73 }
74
75 return true;
76}
